CONTESTNAME=South America 10m Contest
CONTESTWEB=http://sa10m.com.ar
CONTESTRULES=http://sa10m.com.ar/wp/rules

BANDS=10
MODES=CW;SSB
CATEGORY_MODES=CW;SSB;Mixed
CATEGORIES=Single operator;Multi-operator

# Dupe rule
DOUBLE_QSO=PER_MODE

# Prefix is first multiplier
MULT1_TYPE=WPX
MULT1_COUNT=ALL
MULT1_FIELD=FROM_WPX
MULT1_DISPLAY=P

# CQ zone is second multiplier
MULT2_TYPE=CQZONE
MULT2_COUNT=PER_BAND
MULT2_FIELD=RCVD
MULT2_DISPLAY=Z

# Live score multiplier attributes
LIVESCORE_MULT=WPXPREFIX;ZONE

# Scoring
POINTS_TYPE=CALC

# All stations get 2 points for QSO with mobile stations
POINTS_FIELD_BAND_MODE=DEST->CALL:/[AM]M$;ALL;ALL;ALL;2

# All stations get 0 points for QSO with own DXCC entity
POINTS_FIELD_BAND_MODE=SOURCE->DXCC:DEST->DXCC;ALL;ALL;ALL;0

# Mobiles in zone 9-13 gets 2 points for SA stations and 4 points for non SA stations
POINTS_FIELD_BAND_MODE=CONFIG->CALL:/[AM]M$;CONFIG->CQZONE:^(09|1[0123])$;ALL;ALL;2;DEST->CONT:^SA$
POINTS_FIELD_BAND_MODE=CONFIG->CALL:/[AM]M$;CONFIG->CQZONE:^(09|1[0123])$;ALL;ALL;4;!DEST->CONT:^SA$

# Mobiles outside zone 9-13 gets 4 points for SA stations and 2 points non SA stations
POINTS_FIELD_BAND_MODE=CONFIG->CALL:/[AM]M$;!CONFIG->CQZONE:^(09|1[0123])$;ALL;ALL;4;DEST->CONT:^SA$
POINTS_FIELD_BAND_MODE=CONFIG->CALL:/[AM]M$;!CONFIG->CQZONE:^(09|1[0123])$;ALL;ALL;2;!DEST->CONT:^SA$

# Non SA stations gets 4 points for SA stations and 2 points for all other stations
POINTS_FIELD_BAND_MODE=!SOURCE->CONT:^SA$;DEST->CONT:^SA$;ALL;ALL;4
POINTS_FIELD_BAND_MODE=!SOURCE->CONT:^SA$;ALL;ALL;ALL;2

# SA stations gets 4 points for non SA stations and 2 points for all other stations
POINTS_FIELD_BAND_MODE=SOURCE->CONT:^SA$;!DEST->CONT:^SA$;ALL;ALL;4
POINTS_FIELD_BAND_MODE=SOURCE->CONT:^SA$;ALL;ALL;ALL;2

# Summary
